      card.addEventListener('click', () => {
        // Remove selected class from all cards

        card.classList.add('selected');

      modal.style.display = 'none';
      utrInput.value = '';
    });

    // Close modal when clicking outside
    window.addEventListener('click', (e) => {
      if (e.target === modal) {
        modal.style.display = 'none';
        utrInput.value = '';
      }
    });
